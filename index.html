import pygame import sys import random # ---------- 初始化 ---------- pygame.init() WIDTH, HEIGHT = 1000, 300 screen = pygame.display.set_mode((WIDTH, HEIGHT)) pygame.display.set_caption("情侣飞行棋 - Auronix版") # 颜色 WHITE = (255, 255, 255) BLACK = (0, 0, 0) RED = (255, 50, 50) BLUE = (50, 50, 255) GREEN = (50, 255, 50) YELLOW = (255, 255, 50) GRAY = (200, 200, 200) DICE_COLOR = (200, 100, 50) # 字体 font = pygame.font.SysFont("arial", 18) big_font = pygame.font.SysFont("arial", 28) # 棋盘格子 NUM_SQUARES = 50 SQUARE_WIDTH = WIDTH // NUM_SQUARES # 玩家位置 player_pos = {"A": 0, "B": 0} # 奖励/惩罚格 special_squares = { 5: ("奖励：前进2格", "រង្វាន់: ឡើង 2 កន្លែង"), 12: ("惩罚：后退1格", "ទោស: ត្រឡប់ 1 កន្លែង"), 20: ("奖励：前进3格", "រង្វាន់: ឡើង 3 កន្លែង"), 33: ("惩罚：后退2格", "ទោស: ត្រឡប់ 2 កន្លែង"), 42: ("奖励：再掷一次", "រង្វាន់: លេងម្តងទៀត") } # 语言选择 language = "cn" # cn: 中文, kh: 高棉语 # 当前提示信息 message = "" # ---------- 游戏函数 ---------- def dice_jump_animation(final_value): # 骰子初始位置 x, y = WIDTH - 150, 50 velocity = -15 # 弹起速度 gravity = 1 dy = velocity bounce_count = 0 while bounce_count < 2: screen.fill(WHITE) draw_board(dice=None) # 画骰子 pygame.draw.rect(screen, DICE_COLOR, (x, y, 50, 50)) dice_surface = big_font.render(str(final_value), True, BLACK) screen.blit(dice_surface, (x + 15, y + 10)) pygame.display.flip() y += dy dy += gravity if y >= 100: # 碰到底部弹起 y = 100 dy = -dy * 0.6 # 弹力衰减 bounce_count += 1 pygame.time.delay(20) def roll_dice_animation(): for _ in range(10): dice = random.randint(1, 6) draw_board(dice=dice) pygame.time.delay(50) dice = random.randint(1, 6) dice_jump_animation(dice) return dice def move_player(player): global message dice = roll_dice_animation() player_pos[player] = (player_pos[player] + dice) % NUM_SQUARES msg = f"{player} 掷骰子: {dice}, 当前位置: {player_pos[player]}" if player_pos[player] in special_squares: reward = special_squares[player_pos[player]][0] if language=="cn" else special_squares[player_pos[player]][1] msg += " | " + reward if "前进" in reward or "ឡើង" in reward: move = int([s for s in reward if s.isdigit()][0]) player_pos[player] = (player_pos[player] + move) % NUM_SQUARES elif "后退" in reward or "ត្រឡប់" in reward: move = int([s for s in reward if s.isdigit()][0]) player_pos[player] = (player_pos[player] - move) % NUM_SQUARES elif "再掷一次" in reward or "លេងម្តងទៀត" in reward: move_player(player) message = msg print(msg) def draw_board(dice=None): # 画格子 for i in range(NUM_SQUARES): rect = pygame.Rect(i*SQUARE_WIDTH, HEIGHT//2 - 30, SQUARE_WIDTH, 60) color = GRAY if i in special_squares: color = YELLOW pygame.draw.rect(screen, color, rect) pygame.draw.rect(screen, BLACK, rect, 1) # 画玩家 pygame.draw.circle(screen, RED, (player_pos["A"]*SQUARE_WIDTH + SQUARE_WIDTH//2, HEIGHT//2), 15) pygame.draw.circle(screen, BLUE, (player_pos["B"]*SQUARE_WIDTH + SQUARE_WIDTH//2, HEIGHT//2), 15) # 显示提示信息 if message: msg_surface = big_font.render(message, True, BLACK) screen.blit(msg_surface, (10, HEIGHT - 50)) # 显示骰子点数（静态） if dice: pygame.draw.rect(screen, DICE_COLOR, (WIDTH - 150, 50, 50, 50)) dice_surface = big_font.render(str(dice), True, BLACK) screen.blit(dice_surface, (WIDTH - 135, 60)) pygame.display.flip() # ---------- 主循环 ---------- turn_order = ["A", "B"] turn_index = 0 clock = pygame.time.Clock() while True: clock.tick(10) draw_board() for event in pygame.event.get(): if event.type == pygame.QUIT: pygame.quit() sys.exit() # 空格掷骰子 if event.type == pygame.KEYDOWN: if event.key == pygame.K_SPACE: current_player = turn_order[turn_index % 2] move_player(current_player) turn_index += 1 # L键切换语言 if event.key == pygame.K_l: language = "kh" if language=="cn" else "cn" message = f"语言切换为: {'高棉语' if language=='kh' else '中文'}"
